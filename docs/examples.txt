// example using reflect
const edges = svg.makeRange(3, svg.makeEdge, { first: -20, last: 0 }, 200);
const notches = svg.makeRange(2, svg.makeNotch, { first: 75, last: 85 }, 20, 100, 2 / 3);
const side = svg.merge(edges, notches);
const le = svg.reflect(side);
const path = svg.drawPath(10, 200, le);
log.print("edge length:", svg.edgeLength(le));
const leExt = svg.extent(le);
log.print("leading edge extent:", leExt.x, leExt.y);
log.print("done.");

// --- Alfa ---
// leading edge
const leEdges = svg.makeRange(3, svg.makeEdge, { first: -20, last: 0 }, 180);
const leNotches = svg.makeRange(2, svg.makeNotch, { first: 75, last: 85 }, 20, { first: 100, last: 140 }, 2 / 3);
const le = svg.reflect(svg.merge(leEdges, leNotches));
// trailing edge
const teEdges = svg.makeRange(2, svg.makeEdge, { first: 135, last: 180 }, 100);
const teNotches = svg.makeNotch(255, 30, 50, 2 / 3);
const te = svg.reflect(svg.merge(teEdges, teNotches));
// scale trailing edge to match leading edge extent
const scale = svg.extent(le).x / -svg.extent(te).x;
log.print("trailing edge scale factor:", scale);
const teScaled = svg.clone(te, scale);
// draw canopy
const xpos = 600 - svg.extent(le).x / 2; // center at major grid line
svg.drawSurface(xpos, 110, le, teScaled);
//svg.printPlans();
log.print("done.");
